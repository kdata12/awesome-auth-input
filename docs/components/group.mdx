# AuthCode.Group

The root component that manages state and validation for auth code inputs. All `<AuthCode.Input>` components must be children of this component.

```tsx
import { AuthCode } from "awesome-auth-input";

<AuthCode.Group onComplete={(code) => console.log(code)}>
  <AuthCode.Input index={0} />
  <AuthCode.Input index={1} />
  <AuthCode.Input index={2} />
  <AuthCode.Input index={3} />
</AuthCode.Group>;
```

## Props

<ResponseField name="children" type="ReactNode" required>
  The child components, typically `<AuthCode.Input>` components.
</ResponseField>

<ResponseField name="name" type="string">
  A string specifying a name for the input control. This name is submitted along
  with the control's value when the form data is submitted.
</ResponseField>

<ResponseField name="validation" type="ValidationPattern" default='{ type: "numeric" }'>
  Specifies what type of validation the user agent has to provide for each character input.
  
  Options:
  - `{ type: "numeric" }` - Numbers only (0-9)
  - `{ type: "alpha" }` - Letters only (a-z, A-Z)
  - `{ type: "alphanumeric" }` - Letters and numbers
  - `{ type: "custom", regex: RegExp, pattern: string, inputMode: string }` - Custom validation
</ResponseField>

<ResponseField name="autoSubmit" type="boolean" default="true">
  Whether or not the component should attempt to automatically submit when all
  fields are filled. If the field is associated with an HTML `form` element, the
  form's `requestSubmit` method will be called.
</ResponseField>

<ResponseField name="onComplete" type="(value: string) => void">
  Callback fired when all fields are filled. Receives the complete auth code
  value as a string.
</ResponseField>

<ResponseField name="value" type="string">
  The controlled value of the field. When provided, the component operates in
  controlled mode.
</ResponseField>

<ResponseField name="onValueChange" type="(value: string) => void">
  Callback fired whenever the field value changes. Use together with `value` to
  create a controlled component.
</ResponseField>

<ResponseField name="defaultValue" type="string">
  The initial value of the uncontrolled field. When provided without `value`,
  the component operates in uncontrolled mode.
</ResponseField>

<ResponseField name="type" type='"text" | "password"' default='"text"'>
  The type of control to render. When set to `"password"`, the input characters
  will be masked.
</ResponseField>

<ResponseField name="disabled" type="boolean" default="false">
  Whether or not the input elements are disabled.
</ResponseField>

## Validation Patterns

### Numeric (Default)

```tsx
<AuthCode.Group validation={{ type: "numeric" }}>
  {/* Only accepts 0-9 */}
</AuthCode.Group>
```

### Alphabetic

```tsx
<AuthCode.Group validation={{ type: "alpha" }}>
  {/* Only accepts a-z, A-Z */}
</AuthCode.Group>
```

### Alphanumeric

```tsx
<AuthCode.Group validation={{ type: "alphanumeric" }}>
  {/* Accepts letters and numbers */}
</AuthCode.Group>
```

### Custom

```tsx
<AuthCode.Group
  validation={{
    type: "custom",
    regex: /^[A-Z]$/,
    pattern: "[A-Z]{1}",
    inputMode: "text",
  }}
>
  {/* Only accepts uppercase letters */}
</AuthCode.Group>
```

## Controlled vs Uncontrolled

### Uncontrolled (Simple)

```tsx
<AuthCode.Group defaultValue="12" onComplete={(code) => verifyCode(code)}>
  <AuthCode.Input index={0} />
  <AuthCode.Input index={1} />
  <AuthCode.Input index={2} />
  <AuthCode.Input index={3} />
</AuthCode.Group>
```

### Controlled

```tsx
const [code, setCode] = useState('');

<AuthCode.Group
  value={code}
  onValueChange={setCode}
  onComplete={(code) => verifyCode(code)}
>
  <AuthCode.Input index={0} />
  <AuthCode.Input index={1} />
  <AuthCode.Input index={2} />
  <AuthCode.Input index={3} />
</AuthCode.Group>

<button onClick={() => setCode('')}>Clear</button>
```

## Form Integration

The component includes a hidden input for native form submission:

```tsx
<form
  onSubmit={(e) => {
    e.preventDefault();
    const formData = new FormData(e.currentTarget);
    console.log("OTP:", formData.get("otp"));
  }}
>
  <AuthCode.Group name="otp" autoSubmit>
    <AuthCode.Input index={0} />
    <AuthCode.Input index={1} />
    <AuthCode.Input index={2} />
    <AuthCode.Input index={3} />
  </AuthCode.Group>
  <button type="submit">Verify</button>
</form>
```
